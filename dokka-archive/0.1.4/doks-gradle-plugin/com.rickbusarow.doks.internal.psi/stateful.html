internal fun <T> Sequence<T>.stateful(): Sequence<T>(source)

A Sequence which only yields each element once, but is not constrained to only one consumer via constrainOnce.

val seq = sequence {
  repeat(5) { num ->
    println("    yield $num")
    yield(num)
  }
}
  .stateful()

repeat(3) {
  println(seq.firstOrNull { num -> num % 3 == 0 })
